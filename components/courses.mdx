## AyudantiaList

Main container for displaying class sessions in a timeline format with search and filtering.

**Location:** `~/workspace/source/didg-website/src/components/courses/AyudantiaList.tsx`

### Features

- **Search** - Filters by title and description
- **Filter Types** - ALL, PDF, VIDEO
- **Timeline Layout** - Vertical timeline with gradient connector
- **Sticky Search Bar** - Stays visible on scroll
- **Empty States** - Custom message when no results

### Usage Example

```tsx
import { AyudantiaList } from "@/components/courses/AyudantiaList";
import { createClient } from "@/infrastructure/supabase/server";

export default async function SubjectPage({ params }) {
  const supabase = await createClient();
  const { subjectId } = await params;

  // Fetch sessions
  const { data: ayudantias } = await supabase
    .from("ayudantias")
    .select("*")
    .eq("subject_id", subjectId);

  // Fetch user's bookmarks
  const { data: { user } } = await supabase.auth.getUser();
  const bookmarkedSet = new Set<string>();

  if (user) {
    const { data: bookmarks } = await supabase
      .from("ayudantia_bookmarks")
      .select("ayudantia_id")
      .eq("user_id", user.id);

    bookmarks?.forEach((b) => bookmarkedSet.add(b.ayudantia_id));
  }

  return (
    <AyudantiaList
      ayudantias={ayudantias || []}
      subjectName="Database Systems"
      bookmarkedIds={bookmarkedSet}
    />
  );
}
```

const filtered = ayudantias.filter((ayu) => {

### State Logic

```typescript
const [isBookmarked, setIsBookmarked] = useState(initialState);
const [loading, setLoading] = useState(false);

const handleClick = async (e: React.MouseEvent) => {
  e.stopPropagation(); // Prevent card click
  if (loading) return;

  const previousState = isBookmarked;
  setIsBookmarked(!isBookmarked); // Optimistic
  setLoading(true);

  const res = await toggleAyudantiaBookmark(ayudantiaId);
  if (!res.ok) setIsBookmarked(previousState); // Rollback
  setLoading(false);
};
```

### Download Flow

```typescript
const handleDownload = async () => {
  setIsLoading(true);
  try {
    // 1. Get signed URL from server action
    const url = await getSignedUrl(filePath);

    // 2. Track analytics
    const resourceName = fileName || filePath.split("/").pop() || "Archivo";
    trackDownload(resourceName, subjectName);

    // 3. Open in new tab
    window.open(url, "_blank");
  } catch (error) {
    alert("Error al descargar el archivo. Intenta nuevamente.");
  } finally {
    setIsLoading(false);
  }
};
```
