### Implementation

```tsx
import { useTheme } from "next-themes";
import { Moon, Sun } from "lucide-react";

export function ThemeToggle() {
  const { theme, setTheme } = useTheme();
  const [mounted, setMounted] = useState(false);

  useEffect(() => setMounted(true), []);
  if (!mounted) return null; // Avoid hydration mismatch

  return (
    <button
      onClick={() => setTheme(theme === "dark" ? "light" : "dark")}
      className="fixed bottom-6 right-6 z-50 p-3 rounded-full bg-surface border border-primary/20 shadow-[0_0_15px_rgba(0,0,0,0.2)] hover:shadow-[0_0_20px_var(--primary)]"
    >
      <Sun className="h-6 w-6 text-orange-500 rotate-0 scale-100 dark:-rotate-90 dark:scale-0" />
      <Moon className="h-6 w-6 text-primary rotate-90 scale-0 dark:rotate-0 dark:scale-100" />
    </button>
  );
}
```

### Styling Pattern


## GradeSimulator

Interactive calculator for simulating final grades.

**Location:** `~/workspace/source/didg-website/src/components/tools/GradeSimulator.tsx`

### Features

- **Dynamic Evaluations** - Add/remove items
- **Weight Management** - Percentage-based weighting
- **Target Grade** - Calculate required score
- **Partial Average** - Current standing
- **Visual Feedback** - Color-coded results

### Usage Example

```tsx
import { GradeSimulator } from "@/components/tools/GradeSimulator";

export default function GradesPage() {
  return (
    <div className="min-h-screen py-12 px-4">
      <h1>Simulador de Notas</h1>
      <GradeSimulator />
    </div>
  );
}
```

### Data Structure

```typescript
interface GradeItem {
  id: string;
  name: string;       // Evaluation name
  weight: number;     // Percentage (0-100)
  grade: number | null; // Score (1.0-7.0)
}
```

### State Management

```typescript
const [grades, setGrades] = useState<GradeItem[]>([
  { id: "1", name: "Certamen 1", weight: 25, grade: null },
  { id: "2", name: "Certamen 2", weight: 25, grade: null },
  { id: "3", name: "Tareas", weight: 20, grade: null },
]);

const [targetGrade, setTargetGrade] = useState(4.0);
const [needed, setNeeded] = useState<number | null>(null);
const [currentAverage, setCurrentAverage] = useState<number | null>(null);
```

### Calculation Logic

```typescript
useEffect(() => {
  let totalWeightedScore = 0;
  let totalWeightUsed = 0;

  // Calculate current average
  grades.forEach((g) => {
    if (g.grade !== null) {
      totalWeightedScore += g.grade * (g.weight / 100);
      totalWeightUsed += g.weight;
    }
  });

  if (totalWeightUsed > 0) {
    setCurrentAverage(parseFloat((totalWeightedScore / (totalWeightUsed / 100)).toFixed(1)));
  }

  // Calculate needed grade
  const totalDefinedWeight = grades.reduce((sum, g) => sum + g.weight, 0);
  if (totalDefinedWeight < 100) {
    const remaining = 100 - totalDefinedWeight;
    const scoreNeeded = (targetGrade - totalWeightedScore) / (remaining / 100);
    setNeeded(parseFloat(scoreNeeded.toFixed(1)));
  }
}, [grades, targetGrade]);
```

### Result Display

```tsx
// Color-coded feedback
<div
  className={`p-4 rounded-xl text-center ${
    needed && needed > 7.0
      ? "bg-red-500/10 border-red-500/30"
      : needed && needed <= 4.0
      ? "bg-emerald-500/10 border-emerald-500/30"
      : "bg-indigo-500/10 border-indigo-500/30"
  }`}
>
  <p>Para aprobar con {targetGrade} necesitas:</p>
  <span
    className={`text-4xl font-bold ${
      needed > 7.0
        ? "text-red-500"
        : needed > 5.5
        ? "text-orange-500"
        : "text-emerald-500"
    }`}
  >
    {needed}
  </span>
  {needed > 7.0 && <p>💀 F, soldado.</p>}
</div>
```


**Reference:** `~/workspace/source/didg-website/src/app/courses/[subjectId]/page.tsx:51`

### Configuration

```typescript
const AFK_TIME = 120000; // 2 minutes until sleep
const ATTACK_DURATION = 1600; // Attack animation length
const BASE_BOTTOM = 16; // Base distance from bottom

const SPRITES = {
  idle: "/assets/idle.gif",
  sleeping: "/assets/sleep.gif",
  attacking: "/assets/attack.gif",
};
```

### State Logic

```typescript
const [status, setStatus] = useState<"idle" | "sleeping" | "attacking">("idle");
const [bottomOffset, setBottomOffset] = useState(BASE_BOTTOM);

// Activity detection
useEffect(() => {
  const handleActivity = () => {
    if (statusRef.current === "sleeping") setStatus("idle");
    if (timerRef.current) clearTimeout(timerRef.current);

    timerRef.current = setTimeout(() => {
      if (statusRef.current !== "attacking") setStatus("sleeping");
    }, AFK_TIME);
  };

  window.addEventListener("mousemove", handleActivity);
  window.addEventListener("keydown", handleActivity);
  // ... more events
}, []);

// Click interaction
const handleClick = () => {
  if (statusRef.current === "attacking") return;
  setStatus("attacking");
  setTimeout(() => setStatus("idle"), ATTACK_DURATION);
};
```

### Footer Collision Detection

```typescript
useEffect(() => {
  const handleScrollPosition = () => {
    const footer = document.querySelector("footer");
    if (footer) {
      const footerRect = footer.getBoundingClientRect();
      const windowHeight = window.innerHeight;

      if (footerRect.top < windowHeight) {
        const overlap = windowHeight - footerRect.top;
        setBottomOffset(BASE_BOTTOM + overlap);
      } else {
        setBottomOffset(BASE_BOTTOM);
      }
    }
  };

  window.addEventListener("scroll", handleScrollPosition);
  window.addEventListener("resize", handleScrollPosition);
}, []);
```

### Rendering

```tsx
<div
  className="fixed left-2 md:left-4 z-40 cursor-pointer hover:scale-110 active:scale-95"
  style={{ bottom: `${bottomOffset}px` }}
  onClick={handleClick}

  {/* Sleep bubble */}
  {status === "sleeping" && (
    <div className="absolute -top-4 right-0 animate-bounce text-[10px] font-mono text-white bg-black/60 px-1.5 py-0.5 rounded-full">
      Zzz...
    </div>
  )}

  {/* Sprite */}
  <div className="w-14 h-14 sm:w-16 sm:h-16 md:w-20 md:h-20">
    <img
      src={SPRITES[status]}
      alt="Charmander"
      style={{ imageRendering: "pixelated" }}
      draggable={false}
    />
  </div>

  {/* Fire effect */}
  {status === "attacking" && (
    <div className="absolute bottom-0 left-1/2 -translate-x-1/2 w-12 md:w-16 h-4 bg-orange-500/40 blur-lg rounded-full animate-pulse" />
  )}
</div>
```

## Utility: cn()

Conditional className utility for merging Tailwind classes.

**Location:** `~/workspace/source/didg-website/src/core/utils/cn.ts`

### Usage

```tsx
import { cn } from "@/core/utils/cn";

// Merge base and conditional classes
<div
  className={cn(
    "base-class",
    condition && "conditional-class",
    isActive ? "active-class" : "inactive-class",
    className // User override
  )}
/>
```

### Examples

```tsx
// Button variants
className={cn(
  "px-4 py-2 rounded",
  variant === "primary" && "bg-primary text-background",
  variant === "secondary" && "bg-secondary text-white",
  isLoading && "opacity-50 pointer-events-none"
)}

// Responsive sizing
className={cn(
  "w-full",
  size === "sm" && "h-8 text-sm",
  size === "md" && "h-10 text-base",
  size === "lg" && "h-12 text-lg"
)}
```
